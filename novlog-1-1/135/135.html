#SOLVE

무작정 for문을 돌리면 시간제한이 0.15초 이기에 시간초과가 발생한다.

그래서 자리수가 변경될 때 마다 자릿수를 늘리는 방식으로 풀이했다.

#include <iostream>
using namespace std;

int main(){
	int inputNum = 0;
	int ans = 0;
	cin >> inputNum;
	int length = 1, boundary = 1;
	
	for (int i = 1 ; i <= inputNum ; i++) {
		if (i == boundary * 10) {
			boundary *= 10;
			length++;
		}
		ans += length;
	}
	
	cout << ans << endl;
	return 0;
}
