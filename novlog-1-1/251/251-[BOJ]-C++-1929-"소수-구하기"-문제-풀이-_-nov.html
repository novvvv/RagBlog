int sieve[1000001] = {0, };
void eratos(int n, int m){
	for(int i = 2; i <= m; ++i) sieve[i] = i;
	// 2의 배수를 제거한다.
	for(int i = 2; i <= m; ++i){
		// 이미 지워진 숫자는 continue
		if(sieve[i] == 0) continue;
		for(int j = i + i; j <= m; j += i){
			sieve[j] = 0;
		}
	}
	for(int i = n; i <= m; ++i){
		if(sieve[i] != 0) cout << i << '\n';
	}
}
