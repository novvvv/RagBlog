1. sqrt 제곱근 함수를 이용한 풀이

cmath 라이브러리의 sqrt 제곱근 함수를 이용한 풀이이다.

#include <string>
#include <vector>
#include <cmath>
using namespace std;

long long solution(long long n) {
    bool flag = false;
    long long ans = 0;
    long long temp = sqrt(n);
    if(temp * temp == n) flag = true;
    else flag = false;
    if(flag) return (temp + 1) * (temp + 1);
    else return -1;
}

 

2. 함수를 이용하지 않은 풀이

sqrt 제곱근 함수를 이용하지 않은 풀이이다. 입력받은 n까지 for루프를 돌려서 입력받은 값 n이 특정 정수의 제곱인지 판별하였다. 이 때 for 루프의 변수를 int형으로 선언하면 시간초과가 발생하니 유의해야 한다.

#include <string>
#include <vector>
using namespace std;

long long solution(long long n) {
    bool flag = false;
    long long temp = 0;
    for(long long i = 1; i <= n; i++){
        if((i * i) == n){
            flag = true;
            temp = i;
            break;
        }
    }
    if(flag) return (temp + 1) * (temp + 1);
    else return -1;
}
