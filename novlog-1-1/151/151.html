#CODE

#include <iostream>
#include <queue>
using namespace std;

int main(int argc, char* argv[]){
	int N, K;
	cin >> N >> K;
	queue<int> q;
	for (int i = 1 ; i <= N; i++) {
		q.push(i);
	}
	cout << "<";
	int cnt = 1;
	while(!q.empty()) {
		if (cnt % K == 0) {
			int ans = q.front();
			q.pop();
			if (q.empty()) { cout << ans << ">"; }
			else { cout << ans << ", ";}
		}
		else {
			int tmp = q.front();
			q.pop();
			q.push(tmp);
		}
		cnt++;
	}
	return 0;
}
